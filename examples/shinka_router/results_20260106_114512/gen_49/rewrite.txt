"""ShinkaRouter: Adaptive routing agent based on difficulty classification for AIME problems."""
import re
from typing import Callable, Tuple, List
from collections import Counter

class Agent:
    def __init__(
        self,
        query_llm: Callable,
        quick_temp: float = 0.7,
        deep_temp: float = 0.0,
        verify_temp: float = 0.0,
        ensemble_size: int = 3,
    ):
        self.query_llm = query_llm
        self.quick_temp = quick_temp
        self.deep_temp = deep_temp
        self.verify_temp = verify_temp
        self.ensemble_size = ensemble_size
        self._calls = []

        self.output_format = (
            "On the final line output only the digits of the answer (0-999). "
            "Provide your final answer enclosed in a LaTeX \\boxed{{...}} command."
        )

    def _track(self, name: str):
        self._calls.append(name)

    def get_calls(self):
        return self._calls.copy()

    def reset(self):
        self._calls.clear()

    @staticmethod
    def extract_boxed(response: str) -> str:
        idx = response.rfind("\\boxed")
        if idx < 0:
            idx = response.rfind("\\fbox")
        if idx < 0:
            return None
        brace_idx = response.find("{", idx)
        if brace_idx < 0:
            return None
        level = 0
        for i in range(brace_idx, len(response)):
            if response[i] == "{":
                level += 1
            elif response[i] == "}":
                level -= 1
                if level == 0:
                    content = response[brace_idx + 1:i]
                    return content.strip().lstrip("0") or "0"
        return None

    def estimate_difficulty(self, problem: str) -> Tuple[str, float]:
        self._track("estimate_difficulty")
        prompt = (
            f"Analyze this AIME problem and classify its difficulty as one word: easy, medium, or hard.\n\n"
            f"Problem: {problem}"
        )
        response, c = self.query_llm(prompt, system="Expert at difficulty assessment.", temperature=0.0)
        diff = response.strip().lower()
        if "easy" in diff:
            return "easy", c
        elif "hard" in diff:
            return "hard", c
        else:
            return "medium", c

    def baseline_solve(self, problem: str) -> Tuple[str, float]:
        self._track("baseline_solve")
        prompt = f"{self.output_format}\n\n{problem}\n\n"
        return self.query_llm(prompt, system="You are a skilled mathematician.", temperature=0.0)

    def deep_think(self, problem: str) -> Tuple[str, float]:
        self._track("deep_think")
        prompt = (
            f"{self.output_format}\n\n"
            f"Solve this problem carefully with detailed reasoning:\n\n{problem}"
        )
        return self.query_llm(prompt, system="You are an expert mathematician. Think step-by-step.", temperature=self.deep_temp)

    def verify(self, problem: str, response: str) -> Tuple[str, float]:
        self._track("verify")
        prompt = (
            f"{self.output_format}\n\n"
            f"Problem: {problem}\n\n"
            f"Solution:\n{response}\n\n"
            "Verify this solution carefully. If errors are found, correct and revise."
        )
        return self.query_llm(prompt, system="You are a rigorous professor.", temperature=self.verify_temp)

    def ensemble_vote(self, problem: str, n: int = None) -> Tuple[str, float]:
        self._track("ensemble_vote")
        n = n or self.ensemble_size
        prompt = f"{self.output_format}\n\n{problem}\n\n"
        responses = []
        total_c = 0.0
        for _ in range(n):
            resp, c = self.query_llm(prompt, system="You are a skilled mathematician.", temperature=0.5)
            responses.append((resp, c))
            total_c += c
        answers = [self.extract_boxed(r[0]) for r in responses if self.extract_boxed(r[0]) is not None]
        if answers:
            count = Counter(answers)
            winner, _ = count.most_common(1)[0]
            for r, c in responses:
                if self.extract_boxed(r) == winner:
                    return r, total_c
        return responses[0][0], total_c

    def forward(self, problem: str) -> Tuple[str, float]:
        self.reset()
        total_c = 0.0
        # Step 1: estimate difficulty
        diff, c_diff = self.estimate_difficulty(problem)
        total_c += c_diff
        # Route based on difficulty
        if diff == "easy":
            resp, c = self.baseline_solve(problem)
            total_c += c
            return resp, total_c
        elif diff == "medium":
            resp, c = self.deep_think(problem)
            total_c += c
            resp_v, c_v = self.verify(problem, resp)
            total_c += c_v
            return resp_v, total_c
        else:
            resp, c = self.ensemble_vote(problem, n=3)
            total_c += c
            resp_v, c_v = self.verify(problem, resp)
            total_c += c_v
            return resp_v, total_c
